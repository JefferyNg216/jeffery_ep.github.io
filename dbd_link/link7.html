Creating Multi-Dimensional Arrays

import numpy as np

list_2D = [[1,2,3],[4,5,6],[7,8,9]]
mat1 = np.array(list_2D)
print("Type/Class of this object:",type(mat1))
#Result = Type/Class of this object: <class 'numpy.ndarray'>

print("Here is the matrix\n----------\n",mat1,"\n----------")
''Result =
Here is the matrix
----------
[[1 2 3]
[4 5 6]
[7 8 9]]
----------
''

tuple_2D = np.array([(1.5,2,3), (4,5,6)])

The Dimension, Shape, Size, and Data Type of the Two-dimensional Array

import numpy as np
list_2D = [[1,2,3],[4,5,6],[7,8,9]]
mat1 = np.array(list_2D)

print("Dimension of this matrix: ",mat1.ndim,sep='')
#Result = Dimension of this matrix: 2

print("Size of this matrix: ", mat1.size,sep='')
#Result = Size of this matrix: 9

print("Shape of this matrix: ", mat1.shape,sep='')
#Result = Shape of this matrix: (3, 3)

print("Data type of this matrix: ", mat1.dtype,sep='')
#Result = Data type of this matrix: int64
